\subsection{Convex Sets}

\begin{definition}[conv]
	A set $\mathcal{C}$ is convex in and only if for all
	$x,y \in \mathcal{C}$ and $\theta \in [0,1]$:
	$$\theta x + (1-\theta)y \in \mathcal{C}$$
\end{definition}
